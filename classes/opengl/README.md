void setupPalette(HDC hDC)

    PIXELFORMATDESCRIPTOR pfd;

    DescribePixelFormat(hDC, pixelFormat, sizeof(PIXELFORMATDESCRIPTOR), &pfd);

        paletteSize = 1 << pfd.cColorBits;

        return;
